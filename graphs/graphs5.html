#include < stdio.h>
                #include < stdlib.h>
                
                #define MAX_VERTICES 100
                
                // Function to perform breadth-first search traversal
                void BFS(int startVertex, int vertices, int adjacencyMatrix[][MAX_VERTICES], int visited[]) {
                    int queue[MAX_VERTICES];
                    int front = 0, rear = -1;
                
                    // Enqueue the starting vertex and mark it as visited
                    queue[++rear] = startVertex;
                    visited[startVertex] = 1;
                
                    printf("BFS Traversal starting from vertex %d: ", startVertex);
                
                    while (front <= rear) {
                        // Dequeue a vertex and print it
                        int currentVertex = queue[front++];
                        printf("%d ", currentVertex);
                
                        // Enqueue adjacent vertices that are not visited
                        for (int i = 0; i < vertices; ++i) {
                            if (adjacencyMatrix[currentVertex][i] == 1 && !visited[i]) {
                                queue[++rear] = i;
                                visited[i] = 1;
                            }
                        }
                    }
                
                    printf("\n");
                }
                
                int main() {
                    int vertices;
                
                    // Input the number of vertices
                    printf("Enter the number of vertices: ");
                    scanf("%d", &vertices);
                
                    int adjacencyMatrix[MAX_VERTICES][MAX_VERTICES];
                
                    // Input the adjacency matrix
                    printf("Enter the adjacency matrix (%d x %d):\n", vertices, vertices);
                    for (int i = 0; i < vertices; ++i) {
                        for (int j = 0; j < vertices; ++j) {
                            scanf("%d", &adjacencyMatrix[i][j]);
                        }
                    }
                
                    int startVertex;
                
                    // Input the starting vertex for BFS traversal
                    printf("Enter the starting vertex for BFS traversal: ");
                    scanf("%d", &startVertex);
                
                    int visited[MAX_VERTICES] = {0};
                
                    // Perform BFS traversal
                    BFS(startVertex, vertices, adjacencyMatrix, visited);
                
                    return 0;
                }
