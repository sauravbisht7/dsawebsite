#include < stdio.h>
                #include < stdlib.h>
                
                // Function to perform depth-first search traversal
                void DFS(int vertex, int vertices, int adjacencyMatrix[][100], int visited[]) {
                    printf("%d ", vertex);
                    visited[vertex] = 1;
                
                    for (int i = 0; i < vertices; ++i) {
                        if (adjacencyMatrix[vertex][i] == 1 && !visited[i]) {
                            DFS(i, vertices, adjacencyMatrix, visited);
                        }
                    }
                }
                
                // Function to initiate DFS traversal from a specific vertex
                void DFSTraversal(int startVertex, int vertices, int adjacencyMatrix[][100]) {
                    int visited[vertices];
                    for (int i = 0; i < vertices; ++i) {
                        visited[i] = 0;
                    }
                
                    printf("DFS Traversal starting from vertex %d: ", startVertex);
                    DFS(startVertex, vertices, adjacencyMatrix, visited);
                    printf("\n");
                }
                
                int main() {
                    int vertices;
                
                    // Input the number of vertices
                    printf("Enter the number of vertices: ");
                    scanf("%d", &vertices);
                
                    int adjacencyMatrix[100][100]; // Assuming a maximum of 100 vertices
                
                    // Input the adjacency matrix
                    printf("Enter the adjacency matrix (%d x %d):\n", vertices, vertices);
                    for (int i = 0; i < vertices; ++i) {
                        for (int j = 0; j < vertices; ++j) {
                            scanf("%d", &adjacencyMatrix[i][j]);
                        }
                    }
                
                    int startVertex;
                
                    // Input the starting vertex for DFS traversal
                    printf("Enter the starting vertex for DFS traversal: ");
                    scanf("%d", &startVertex);
                
                    // Perform DFS traversal
                    DFSTraversal(startVertex, vertices, adjacencyMatrix);
                
                    return 0;
                }
