#include < stdio.h>
                #include < stdlib.h>
                #include < string.h>
                #include < sys/time.h>
                
                // Function to partition the array and return the pivot index
                int partition(char **arr, int low, int high) {
                    char *pivot = arr[high];
                    int i = low - 1;
                
                    for (int j = low; j <= high - 1; j++) {
                        if (strcmp(arr[j], pivot) <= 0) {
                            i++;
                            // Swap arr[i] and arr[j]
                            char *temp = arr[i];
                            arr[i] = arr[j];
                            arr[j] = temp;
                        }
                    }
                
                    // Swap arr[i+1] and arr[high] (pivot)
                    char *temp = arr[i + 1];
                    arr[i + 1] = arr[high];
                    arr[high] = temp;
                
                    return i + 1;
                }
                
                // Function to perform quicksort using iteration
                void quickSort(char **arr, int low, int high) {
                    int stack[high - low + 1];
                    int top = -1;
                
                    // Push initial values of low and high to the stack
                    stack[++top] = low;
                    stack[++top] = high;
                
                    // Keep popping from the stack while it's not empty
                    while (top >= 0) {
                        // Pop high and low
                        high = stack[top--];
                        low = stack[top--];
                
                        // Set pivot element at its correct position in the sorted array
                        int pivotIndex = partition(arr, low, high);
                
                        // If there are elements on the left side of pivot, push them to the stack
                        if (pivotIndex - 1 > low) {
                            stack[++top] = low;
                            stack[++top] = pivotIndex - 1;
                        }
                
                        // If there are elements on the right side of pivot, push them to the stack
                        if (pivotIndex + 1 < high) {
                            stack[++top] = pivotIndex + 1;
                            stack[++top] = high;
                        }
                    }
                }
                
                // Function to print an array of strings
                void printArray(char **arr, int size) {
                    for (int i = 0; i < size; i++)
                        printf("%s ", arr[i]);
                    printf("\n");
                }
                
                int main() {
                    int n;
                
                    // Input the number of strings
                    printf("Enter the number of strings: ");
                    scanf("%d", &n);
                
                    char **arr = (char **)malloc(n * sizeof(char *));
                
                    // Input the strings
                    printf("Enter %d strings:\n", n);
                    for (int i = 0; i < n; i++) {
                        arr[i] = (char *)malloc(100 * sizeof(char));
                        scanf("%s", arr[i]);
                    }
                
                    // Print the original array
                    printf("Original array: ");
                    printArray(arr, n);
                
                    // Measure time before sorting
                    struct timeval start, end;
                    gettimeofday(&start, NULL);
                
                    // Perform quicksort
                    quickSort(arr, 0, n - 1);
                
                    // Measure time after sorting
                    gettimeofday(&end, NULL);
                
                    // Print the sorted array
                    printf("Sorted array: ");
                    printArray(arr, n);
                
                    // Calculate and print the execution time
                    double time_taken = (end.tv_sec - start.tv_sec) * 1e6 + (end.tv_usec - start.tv_usec);
                    time_taken /= 1e6; // convert to seconds
                    printf("Time taken to execute the quicksort: %f seconds\n", time_taken);
                
                    // Free allocated memory
                    for (int i = 0; i < n; i++)
                        free(arr[i]);
                    free(arr);
                
                    return 0;
                }
