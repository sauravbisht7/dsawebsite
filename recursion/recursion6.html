#include < stdio.h>

                // Function to find the smallest element in an array using recursion
                int findSmallest(int arr[], int size) {
                    // Base case: if the array has only one element, return it
                    if (size == 1) {
                        return arr[0];
                    } else {
                        // Recursive case: find the smallest element in the rest of the array
                        int minInRest = findSmallest(arr + 1, size - 1);
                
                        // Compare the smallest element in the rest with the current element
                        return (arr[0] < minInRest) ? arr[0] : minInRest;
                    }
                }
                
                int main() {
                    // Input the size of the array
                    int size;
                    printf("Enter the size of the array: ");
                    scanf("%d", &size);
                
                    // Input the elements of the array
                    int arr[size];
                    printf("Enter %d elements of the array:\n", size);
                    for (int i = 0; i < size; ++i) {
                        scanf("%d", &arr[i]);
                    }
                
                    // Find and print the smallest element in the array
                    int smallest = findSmallest(arr, size);
                    printf("The smallest element in the array is: %d\n", smallest);
                
                    return 0;
                }
