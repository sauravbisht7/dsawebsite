#include < stdio.h>
                #include < stdlib.h>
                
                // Structure to represent a node in a binary tree
                struct TreeNode {
                    int data;
                    struct TreeNode *left;
                    struct TreeNode *right;
                };
                
                // Function to create a new node in the binary tree
                struct TreeNode *createNode(int data) {
                    struct TreeNode *newNode = (struct TreeNode *)malloc(sizeof(struct TreeNode));
                    newNode->data = data;
                    newNode->left = newNode->right = NULL;
                    return newNode;
                }
                
                // Function to create an almost complete binary tree from user input
                struct TreeNode *createAlmostCompleteBinaryTree() {
                    int data;
                    printf("Enter the data for the root node: ");
                    scanf("%d", &data);
                
                    struct TreeNode *root = createNode(data);
                
                    // Queue to keep track of the nodes to be filled
                    struct Queue {
                        struct TreeNode **array;
                        int front, rear, capacity;
                    };
                
                    struct Queue *queue = (struct Queue *)malloc(sizeof(struct Queue));
                    queue->capacity = 100; // Assuming a maximum of 100 nodes
                    queue->array = (struct TreeNode **)malloc(sizeof(struct TreeNode *) * queue->capacity);
                    queue->front = queue->rear = 0;
                
                    // Enqueue the root node
                    queue->array[queue->rear++] = root;
                
                    while (1) {
                        struct TreeNode *current = queue->array[queue->front++];
                
                        printf("Do you want to add a left child to node %d? (1 for yes, 0 for no): ", current->data);
                        int addLeftChild;
                        scanf("%d", &addLeftChild);
                
                        if (addLeftChild) {
                            printf("Enter the data for the left child: ");
                            scanf("%d", &data);
                            current->left = createNode(data);
                            queue->array[queue->rear++] = current->left;
                        }
                
                        printf("Do you want to add a right child to node %d? (1 for yes, 0 for no): ", current->data);
                        int addRightChild;
                        scanf("%d", &addRightChild);
                
                        if (addRightChild) {
                            printf("Enter the data for the right child: ");
                            scanf("%d", &data);
                            current->right = createNode(data);
                            queue->array[queue->rear++] = current->right;
                        }
                
                        // Break if the queue is empty
                        if (queue->front == queue->rear) {
                            break;
                        }
                    }
                
                    free(queue->array);
                    free(queue);
                
                    return root;
                }
                
                // Function to perform in-order traversal of a binary tree
                void inOrderTraversal(struct TreeNode *root) {
                    if (root != NULL) {
                        inOrderTraversal(root->left);
                        printf("%d ", root->data);
                        inOrderTraversal(root->right);
                    }
                }
                
                // Function to free the memory allocated for a binary tree
                void freeBinaryTree(struct TreeNode *root) {
                    if (root != NULL) {
                        freeBinaryTree(root->left);
                        freeBinaryTree(root->right);
                        free(root);
                    }
                }
                
                int main() {
                    // Create an almost complete binary tree based on user input
                    struct TreeNode *root = createAlmostCompleteBinaryTree();
                
                    // Display the in-order traversal of the created binary tree
                    printf("In-order Traversal of the Almost Complete Binary Tree: ");
                    inOrderTraversal(root);
                    printf("\n");
                
                    // Free the allocated memory for the binary tree
                    freeBinaryTree(root);
                
                    return 0;
                }
