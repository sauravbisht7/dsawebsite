#include < stdio.h>
                #include < stdlib.h>
                
                // Structure to represent a binary search tree node
                struct TreeNode {
                    int data;
                    struct TreeNode *left;
                    struct TreeNode *right;
                };
                
                // Function to create a new binary search tree node
                struct TreeNode *createNode(int data) {
                    struct TreeNode *newNode = (struct TreeNode *)malloc(sizeof(struct TreeNode));
                    newNode->data = data;
                    newNode->left = newNode->right = NULL;
                    return newNode;
                }
                
                // Function to construct a binary search tree from pre-order traversal
                struct TreeNode *constructBSTFromPreorder(int preOrder[], int n, int *preIndex, int min, int max) {
                    // Base case: If all elements are processed or current element is not within the valid range for a BST
                    if (*preIndex == n || preOrder[*preIndex] < min || preOrder[*preIndex] > max) {
                        return NULL;
                    }
                
                    // Extract the current element and increment pre-order index
                    int value = preOrder[*preIndex];
                    (*preIndex)++;
                
                    // Recursively construct left and right subtrees
                    struct TreeNode *root = createNode(value);
                    root->left = constructBSTFromPreorder(preOrder, n, preIndex, min, value);
                    root->right = constructBSTFromPreorder(preOrder, n, preIndex, value, max);
                
                    return root;
                }
                
                // Function to perform in-order traversal of a binary search tree
                void inOrderTraversal(struct TreeNode *root) {
                    if (root != NULL) {
                        inOrderTraversal(root->left);
                        printf("%d ", root->data);
                        inOrderTraversal(root->right);
                    }
                }
                
                int main() {
                    // Example pre-order traversal of a BST
                    int preOrder[] = {40, 30, 35, 80, 100};
                    int n = sizeof(preOrder) / sizeof(preOrder[0]);
                
                    // Initialize index for pre-order array
                    int preIndex = 0;
                
                    // Construct the binary search tree from pre-order traversal
                    struct TreeNode *root = constructBSTFromPreorder(preOrder, n, &preIndex, INT_MIN, INT_MAX);
                
                    // Perform in-order traversal to verify the constructed BST
                    printf("In-order Traversal of Constructed BST: ");
                    inOrderTraversal(root);
                    printf("\n");
                
                    // Free the allocated memory for the binary search tree
                    free(root);
                
                    return 0;
                }
