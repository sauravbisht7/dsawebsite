                int solution(int ary[],int size,int ans)
                {
                    if(size< 2)
                    {
                        return -1;
                    }
                    else
                    {
                    // things we need in this program.
                    int lmax[size],rmax[size],i,temp,smaller=0;
                    lmax[0]=ary[0];
                    rmax[size-1]=ary[size-1];
                
                    //let us find the array in which the building with greater floors exists. but from, left to right
                
                    for ( i = 1; i < size; i++)
                    {
                        if(ary[i]>lmax[i-1])
                        temp=ary[i];
                        else
                        temp=lmax[i-1];
                        lmax[i]=temp;
                    }
                
                        //let us find the array in which the building with greater floors exists. but from, right to left
                    //loop will also work bakwards. Point to remember
                
                      for ( i = size-2; i >= 0; i--) //Question : why is size -2??
                      // we cannot compare last value meaning index == size-1. so size-2 will come before size-1 
                    {
                        if(ary[i]>rmax[i+1])
                        temp=ary[i];
                        else
                        temp=rmax[i+1];
                        rmax[i]=temp;
                    }
                
                
                    //let us compare the both new array. Left max = lmax ------ and Right max = rmax
                    //our aim is to find out which building is smaller.
                    //and then, we will reduce the building size to get the actual number of places which holds water. 
                    for ( i = 0; i < size; i++)
                    {
                        if(lmax[i]< rmax[i])
                        smaller= (lmax[i]-ary[i]);   
                        else
                        smaller= (rmax[i]-ary[i]); 
                
                        ans = ans+smaller;  
                    }
                    return ans;
                    }
                    
                }
                void main()
