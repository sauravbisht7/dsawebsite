                        int main()
                        {
                            int array1[20],i,j,k,size;
                            printf("Define the size of your array : ");

                            for ( i = 0; i < size; i++)
                            {
                                printf("\nEnter %d value : ",i+1);
                                scanf("%d",&array1[i]);
                            }
                        
                                    printf("Original array : \n");
                        
                        
                            for ( i = 0; i < size; i++)
                            {
                                printf("%d , ",array1[i]);
                            }
                        //loop to find duplicate values and delete them
                               for ( i = 0; i< size; i++)
                            {
                                for (j  = i+1; j < size; j++)
                                {
                                    if(array1[i]==array1[j])
                                    {
                                        for ( k = j; k < size-1; k++)
                                        {
                                            array1[k]=array1[k+1];
                                        }
                                        //decreases the size of array after removing any duplicate value
                                        size--;
                                        //if the position of element is changed,don't increase the index j
                                        j--;
                                    }
                                }
                                
                            }
                                        printf("\nUpdated array : \n");
                        
                              for ( i = 0; i < size; i++)
                            {
                                printf("%d , ",array1[i]);
                            }

                        }
